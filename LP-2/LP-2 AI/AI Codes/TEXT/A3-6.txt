class Graph:
    def __init__(self, vertices):
        self.V = vertices
        self.graph = []

    def addEdge(self, u, v, w):
        self.graph.append([u, v, w])

    def find(self, parent, i):                            
        if parent[i] != i:
            parent[i] = self.find(parent, parent[i])
        return parent[i]

    # ✅ Fixed: Added `self` as the first parameter (since this is a class method)
    def union(self, parent, rank, x, y):
        xroot = self.find(parent, x)
        yroot = self.find(parent, y)

        if rank[xroot] < rank[yroot]:
            parent[xroot] = yroot
        elif rank[xroot] > rank[yroot]:  # ✅ Use elif to avoid running the else block unnecessarily
            parent[yroot] = xroot
        else:
            parent[yroot] = xroot
            rank[xroot] += 1

    def KruskalMST(self):
        result = []  # Store final MST edges
        i, e = 0, 0  # i for sorted edges, e for edges in MST

        # Step 1: Sort edges by weight
        self.graph = sorted(self.graph, key=lambda item: item[2])

        parent = [i for i in range(self.V)]
        rank = [0] * self.V

        # Step 2: Pick the smallest edge and check for cycle using Union-Find
        while e < self.V - 1:
            u, v, w = self.graph[i]
            i += 1
            x = self.find(parent, u)
            y = self.find(parent, v)

            if x != y:
                result.append([u, v, w])
                e += 1
                # ✅ FIXED THIS LINE:
                # ❌ self.union(self, parent, x, y)
                # ✅ Correct usage:
                self.union(parent, rank, x, y)

        print("Edges in the MST:")
        total = 0
        for u, v, weight in result:
            print(f"{u} -- {v} == {weight}")
            total += weight
        print(f"Minimum Cost: {total}")
